Ejemplo 2: Contar la Frecuencia de Palabras (Java)

Este programa cuenta cu치ntas veces aparece cada palabra en una cadena.

import java.util.HashMap; import java.util.Scanner; public class ContarPalabras { public static void main(String[] args) { Scanner scanner = new Scanner(System.in); System.out.println("Ingresa una frase:"); String texto = scanner.nextLine(); String[] palabras = texto.toLowerCase().split("\\s+"); HashMap frecuencia = new HashMap<>(); for (String palabra : palabras) { frecuencia.put(palabra, frecuencia.getOrDefault(palabra, 0) + 1); } System.out.println("Frecuencia de palabras:"); for (String palabra : frecuencia.keySet()) { System.out.println(palabra + ": " + frecuencia.get(palabra)); } scanner.close(); } }
